སྦིར་དང་ཅཱོན་ཏ་ར་ས་གནས་མི་མང་ཡོངས་ཀྱི་སྙན་ལམ་དུ།

ཆེད་ཞུ།

ས་གནས་ཀྱི་གནད་ཡོད་མི་སྣ་ཁག་དང་བཀའ་མོལ་བྱུང་དོན་བཞིན། རྫོང་སར་བཤད་གྲྭ་ཆོས་ཀྱི་བློ་གྲོས་ཀྱིས་ཉེ་འཁོར་ས་གནས་མི་མང་ལ་དམིགས་ཏེ། མི་མང་རྣམས་གཞུང་ལུགས་ཆེན་པོར་སློབ་གཉེར་གནང་མ་ཐུབ་ཀྱང་ནང་ཆོས་ཀྱི་ལྟ་བ་དང་འགྲོ་སྟངས་ཤེས་པ། སོ་སོས་ཆོས་ཉམས་སུ་ལེན་སྟངས་ཤེས་པ། རང་གི་བུ་ཕྲུག་ལ་ཡང་ལམ་སྟོན་བྱེད་ཤེས་པ་ཡོང་ཕྱིར། བཤད་གྲྭའི་གུང་སེང་ཆེན་མོ་དང་མི་མང་རྣམས་སུད་ཚོང་ལ་འགྲོ་བའི་སྐབས་ཕུད། ཟླ་བ་རེར་ཐེངས་རེ་གནང་རྒྱུའི་ **“****ནང་ཆོས་ངོ་སྤྲོད་****”** ཀྱི་ལས་རིམ་ཞིག་ཟླ་བ་འདི་ནས་བཟུང་སྟེ་དབུ་འཛུགས་ཞུ་རྒྱུ་ཡིན།

ནང་ཆོས་ངོ་སྤྲོད་ཐེངས་དང་པོ་གནང་མཁན་༸སྐྱབས་རྗེ་གདུང་སྲས་ཨ་བི་ཀྲྀ་ཏ་བཛྲ་རིན་པོ་ཆེར་གསོལ་བ་གཏབ་པ་བཞིན་ཞལ་བཞེས་བཟང་པོ་ཐོབ་སྟེ། སྤྱི་ལོ་ 2018 ཟླ་བ་ 8 ཚེས་ 19 གཟའ་ཉི་མ་ཉིན་ནང་ཆོས་ངོ་སྤྲོད་ཐེངས་དང་པོ་གནང་རྒྱུ་ལགས་པས། ས་གནས་མི་མང་རྣམས་ཀྱིས་དེ་དོན་བཞིན་ཕེབས་གནང་ཡོང་བ་ཞུ།

ནང་ཆོས་ངོ་སྤྲོད་སྐབས་དང་པོ།

ངོ་སྤྲོད་གནང་རྒྱུ།        ནང་ཆོས།

ངོ་སྤྲོད་གནང་ཡུལ།     ཅཱོན་ཏ་ར་རྫོང་སར་བཤད་གྲྭའི་འདུ་ཁང།

ངོ་སྤྲོད་གནང་མཁན།    ༸སྐྱབས་རྗེ་གདུང་སྲས་ཨ་བི་ཀྲྀ་ཏ་བཛྲ་རིན་པོ་ཆེ།

ངོ་སྤྲོད་གནང་དུས།      སྤྱི་ལོ་ 2018 ཟླ་བ་ 8 ཚེས་ 19 ། ཞོགས་པའི་ཆུ་ཚོད་ 10 ནས།

རྫོང་སར་བཤད་གྲྭ་ཆོས་ཀྱི་བློ་གྲོས་ཡིག་ཚང་ནས།

སྤྱི་ལོ་ ༢༠༡༨ ཟླ་བ་ ༨ ཚེས་ ༡༠ ཉིན།
