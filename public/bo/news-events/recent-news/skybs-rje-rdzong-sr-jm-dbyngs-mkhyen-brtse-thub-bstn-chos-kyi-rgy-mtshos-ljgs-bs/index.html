༄༅། །༸སྐྱབས་རྗེ་རྫོང་སར་འཇམ་དབྱངས་མཁྱེན་བརྩེ་ཐུབ་བསྟན་ཆོས་ཀྱི་རྒྱ་མཚོས་ལྗགས་བཤད་གནང་བའི་ཕུན་ཚོགས་ལྔ་ལྡན་གྱི་བཤད་པ།

·
    • སྤྱི་ལོ་ ༢༠༢༢ ཕྱི་ཟླ་ ༡༡ ཚེས་ ༢༤ དང་། རབ་བྱུང་ ༡༧ ཆུ་ཕོ་སྟག་གི་ལོའི་བོད་ཟླ་ ༡༠ ཚེས་༡ཉིན་གནས་མཆོག་རྡོ་རྗེ་གདན་དུ་སྡེ་བཞི་དགེ་བཅུ་ཚང་བའི་ཆོས་རྒྱལ་ཆེན་པོའི་འཁོར་འབངས་སུ་གཏོགས་པའི་དཔལ་བརྩེ་བ་ཆེན་པོའི་རྗེས་འཇུག་པ་ཡོངས་ཀྱིས་༸སྐྱབས་མགོན་ས་སྐྱ་གོང་མ་ཁྲི་ཆེན་རྡོ་རྗེ་འཆང་ཨ་ཡུ་བཛྲ་རིན་པོ་ཆེ་མཆོག་ལ་ལོ་རྒྱུས་རང་བཞིན་ཅན་གྱི་བརྟན་བཞུགས་འབུལ་བའི་སྐབས་སུ། སྐྱབས་རྗེ་རྫོང་སར་འཇམ་དབྱངས་མཁྱེན་བརྩེ་ཐུབ་བསྟན་ཆོས་ཀྱི་རྒྱ་མཚོས་ལྗགས་བཤད་གནང་བའི་ཕུན་ཚོགས་ལྔ་ལྡན་གྱི་བཤད་པ་དད་པའི་སྤུ་ལོང་གཡོ་བའི་གསུང་དབྱངས་ཡན་ལག་ལྔ་ལྡན་ཞེས་བྱ་བ་འདི། ལྗགས་བཤད་གནང་བའི་སྒྲ་སྒམ་དང་གཟུགས་མཐོང་བརྙན་ཕྲིན་ལས་ཡིག་ཐོག་ཏུ་ཕབ། མིང་ཡང་བཏགས་ནས་རིན་པོ་ཆེ་མཆོག་ལ་ཞུ་དག་དང་གནང་བ་བཅས་ཞུས་ཏེ་སྤེལ་བ་ཡིན་པས། དཔལ་ལྡན་བླ་མ་དམ་པའི་སྐུ་ཚེ་ཞབས་པད་རྒྱལ་བ་ཚེ་མཐའ་ཡས་པའི་ངོ་བོར་བརྟན་པའི་རྒྱུར་གྱུར་ཅིག་ཅེས་ཡི་གེར་ཕབ་སྒྲིག་པ་གཟི་བརྗིད་མཐའ་ཡས་ཀྱིས་རྫོང་སར་བཤད་གྲྭ་ཆོས་ཀྱི་བློ་གྲོས་ནས་སྨོན་པའོ།། །།སརྦ་མངྒ་ལཾ།། །།
    • སྤྱི་ལོ་ ༢༠༢༢ ཕྱི་ཟླ་ ༡༡ ཚེས་ ༢༤ དང་། རབ་བྱུང་ ༡༧ ཆུ་ཕོ་སྟག་གི་ལོའི་བོད་ཟླ་ ༡༠ ཚེས་༡ཉིན་གནས་མཆོག་རྡོ་རྗེ་གདན་དུ་སྡེ་བཞི་དགེ་བཅུ་ཚང་བའི་ཆོས་རྒྱལ་ཆེན་པོའི་འཁོར་འབངས་སུ་གཏོགས་པའི་དཔལ་བརྩེ་བ་ཆེན་པོའི་རྗེས་འཇུག་པ་ཡོངས་ཀྱིས་༸སྐྱབས་མགོན་ས་སྐྱ་གོང་མ་ཁྲི་ཆེན་རྡོ་རྗེ་འཆང་ཨ་ཡུ་བཛྲ་རིན་པོ་ཆེ་མཆོག་ལ་ལོ་རྒྱུས་རང་བཞིན་ཅན་གྱི་བརྟན་བཞུགས་འབུལ་བའི་སྐབས་སུ། སྐྱབས་རྗེ་རྫོང་སར་འཇམ་དབྱངས་མཁྱེན་བརྩེ་ཐུབ་བསྟན་ཆོས་ཀྱི་རྒྱ་མཚོས་ལྗགས་བཤད་གནང་བའི་ཕུན་ཚོགས་ལྔ་ལྡན་གྱི་བཤད་པ་དད་པའི་སྤུ་ལོང་གཡོ་བའི་གསུང་དབྱངས་ཡན་ལག་ལྔ་ལྡན་ཞེས་བྱ་བ་འདི། ལྗགས་བཤད་གནང་བའི་སྒྲ་སྒམ་དང་གཟུགས་མཐོང་བརྙན་ཕྲིན་ལས་ཡིག་ཐོག་ཏུ་ཕབ། མིང་ཡང་བཏགས་ནས་རིན་པོ་ཆེ་མཆོག་ལ་ཞུ་དག་དང་གནང་བ་བཅས་ཞུས་ཏེ་སྤེལ་བ་ཡིན་པས། དཔལ་ལྡན་བླ་མ་དམ་པའི་སྐུ་ཚེ་ཞབས་པད་རྒྱལ་བ་ཚེ་མཐའ་ཡས་པའི་ངོ་བོར་བརྟན་པའི་རྒྱུར་གྱུར་ཅིག་ཅེས་ཡི་གེར་ཕབ་སྒྲིག་པ་གཟི་བརྗིད་མཐའ་ཡས་ཀྱིས་རྫོང་སར་བཤད་གྲྭ་ཆོས་ཀྱི་བློ་གྲོས་ནས་སྨོན་པའོ།། །།སརྦ་མངྒ་ལཾ།། །།

དེ་གང་འདྲ་བྱས་ནས་ཡོང་གི་ཡོད་རེད་ཟེར་ན། སྐབས་འདི་ལ་ཕུན་སུམ་ཚོགས་པ་ཞེས་པ་དེ་ཟས་དང་གོས། ནོར་དང་ལོངས་སྤྱོད། ཆབ་སྲིད་དང་དབང་ཆ་ལ་སོགས་པ་ཚེ་འདིའི་དང་། ཚེ་འདི་ཡི་ནང་ནས་ཀྱང་འཇིག་རྟེན་གྱི་ཕུན་སུམ་ཚོགས་པ་ལ་བཤད་རྒྱུ་མ་ཡིན་པར། མ་རིག་པའི་གཉིད་ལས་སངས་ཐུབ་པའི་རྒྱུ་དང་རྐྱེན་དེ་རྣམས་ཚུར་ཚོགས་ཡོད་པ་ཞིག་ལ་ཕུན་སུམ་ཚོགས་པ་ཞེས་བཤད་རྒྱུ་ཡིན་པ་དང་། ཡང་ན་ནི་ཤེས་བྱ་ལ་བློ་གྲོས་ཀྱི་པདྨོ་རྒྱས་པའི་ཆ་རྐྱེན་མཚམས་མཚམས་ལ་ཚུར་འཛོམས་རྒྱུ་ཡོད་པ་དེ་འདྲ་ཞིག་ལ་ཕུན་སུམ་ཚོགས་པ་ཞེས་པའི་ཐ་སྙད་འདི་འཇུག་གི་ཡོད་པ་ཡིན་དུས། དང་པོ་ཕུན་སུམ་ཚོགས་པ་གཞན་རྣམས་གང་དུ་ཚོགས་པའི་གནས་ཕུན་སུམ་ཚོགས་པའི་ནི། དེ་ཡང་ད་ལྟ་གོང་དུ་ཞུས་པ་ནང་བཞིན་དོན་དམ་པའི་དབང་དུ་བྱས་པ་ཡིན་ན། འཇིག་རྟེན་གྱི་ཁམས་ལ་ཚད། གྲངས། མཐོ་དམན། གོ་རིམ། སྟེང་དང་འོག །འཛམ་བུ་གླིང་རྡོག་རྡོག་དང་། འཛམ་བུ་གླིང་ལེབ་ལེབ་རེད་ཅེས་པ་དེ་འདྲ་དང་། ཁ་དོག་དཀར་དམར་ཞེས་པ་དེ་འདྲ་གང་ཡང་གྲུབ་ཡོད་ས་མ་རེད། ཡིན་ནའང་བློ་ཀུན་རྫོབ་པ་ཚུར་མཐོང་སོ་སོའི་སྐྱེ་བོ་མོས་པ་དང་། ཁམས་དང་། དབང་པོ་དང་། སྣང་བ་སྣ་ཚོགས་པ་དེ་འདྲ་མང་པོ་ཡོད་དུས། འཇིག་རྟེན་གྱི་ཁམས་ཀྱང་སྣ་ཚོགས་པ་དེ་འདྲ་ཡོད་ས་རེད། འཇིག་རྟེན་ཁམས་དེ་ཚོ་ཡང་ག་ཤེད་ཐག་རིང་པོ་བར་སྣང་སྟོང་པ་ཞིག་ལ་ཡོད་བསྡད་པ་མ་ཡིན་པར། ད་ལྟ་ང་རང་ཚོས་ཇ་འཐུང་ས་ཕོར་པ་གཅིག་གི་ནང་ལ་ཡང་། འཇིག་རྟེན་ཁམས་སྣ་ཚོགས་པ་ཡོད་རེད་ཅེས་མདོ་རྒྱུད་བསྟན་བཅོས་མང་པོའི་ནང་དུ་དེ་འདྲ་གསུངས་ཡོད་ས་རེད།

ད་ལྟ་ང་རང་ཚོའི་སངས་རྒྱས་ཤཱཀྱ་ཐུབ་པ་དེ་སྤྲུལ་པའི་སྐུ་ཡིན་ཞེས་དེ་འདྲ་གསུང་གི་ཡོད་རེད། སྤྲུལ་པའི་སྐུ་སངས་རྒྱས་ཤཱཀྱ་ཐུབ་པ་དེ་ལོངས་སྤྱོད་རྫོགས་སྐུའི་རྣམ་པ་ཅན་དུ་བཞུགས་དུས། སངས་རྒྱས་རྣམ་པར་སྣང་མཛད་ཅེས་གསུང་གི་ཡོད་རེད། དེ་འདྲ་བའི་སངས་རྒྱས་རྣམ་པར་སྣང་མཛད་ཀྱི་སྐུ། གསུང་། ཐུགས། ཡོན་ཏན། ཕྲིན་ལས། ཞིང་ཁམས་ཞེས་པ་དེ་ཚོ། ང་ཚོ་འདྲ་བའི་གདུལ་བྱའི་བློ་ངོ་ལ་ཤོང་ཐུབ་པ་ཤིན་ཏུ་དཀའ་ལས་ཁག་པོ་ཡིན་ས་རེད། ཡིན་ནའང་དཔེར་ན་སངས་རྒྱས་རྣམ་པར་སྣང་མཛད་ཀྱི་སྐུ་ཡི་ཡོན་ཏན་སྐོར་ལ་བསམ་བློ་གཏོང་བའི་ཚུལ་འདྲ་ཞིག་བྱས་གཞག་པ་ཡིན་ན། སངས་རྒྱས་རྣམ་པར་སྣང་མཛད་ཅེས་པའི་སྐུ་ཡི་བ་སྤུ་གཅིག་ལ་ཡང་ཞིང་ཁམས་ཐམས་ཅད་ཀྱིས་ཁྱབ་ནས་ཡོད་པ་དང་། སངས་རྒྱས་དང་བྱང་ཆུབ་སེམས་དཔའ་མང་པོའི་ཞིང་ཁམས་ཐམས་ཅད་ལ། སངས་རྒྱས་རྣམ་པར་སྣང་མཛད་ཀྱི་སྐུ་ཡི་བ་སྤུ་གཅིག་གིས་ཕར་ཁྱབ་ཡོད་པ། དཔེར་ན་སངས་རྒྱས་རྣམ་པར་སྣང་མཛད་ཀྱི་ཕྱག་གཉིས་མཉམ་གཞག་གི་ནང་ལ་ལྷུང་གཟེད་ཅིག་ཡོད་པ། ལྷུང་གཟེད་དེའང་སྤོས་ཀྱི་ཆུ་ཡིས་གང་ནས་ཡོད་པ། སྤོས་ཀྱི་ཆུ་དེའང་ད་ལྟ་ང་རང་ཚོས་ཆུ་འཐུང་སའི་ཕོར་པ་ཆུང་ཆུང་གི་ནང་ལ་ཡོད་པའི་ཆུ་དེ་འདྲ་མ་ཡིན་པར། འཛམ་བུ་གླིང་ན་ཡོད་པའི་མཐའ་ཡས་རྒྱ་མཚོ་ཆེན་པོ་ཐམས་ཅད་ཚུར་བསྡུས་གཞག་ནའང་། དེ་ལས་འབུམ་ཕྲག་མང་པོས་ཀྱང་ལྷག་པའི་མཐའ་ཡས་པའི་རྒྱ་མཚོ་བསམ་གྱིས་མི་ཁྱབ་པ་དེ་འདྲ་ཡོད་པ། དེ་འདྲ་བའི་སྤོས་ཆུ་ཆེན་པོ་དེའི་ནང་དེ་ན། མེ་ཏོག་པདྨའི་སྡོང་པོ་ཡངས་ཤིང་རྒྱ་ཆེན་བ་ཞིག་ཡོད་པའི་མེ་ཏོག་པདྨའི་སྡོང་པོ་དེའི་སྟེང་དེ་ལ། མེ་ཏོག་པདྨ་ཉི་ཤུ་རྩ་ལྔ་གཅིག་གི་སྟེང་དུ་གཅིག་བརྩེགས་ནས་ཡོད་པ། མེ་ཏོག་པདྨ་རེ་རེའི་སྟེང་ལ་ཡང་འཇིག་རྟེན་གྱི་ཁམས་སྣ་ཚོགས་པ་ཡོད་པ། དབུས་ཀྱི་མེ་ཏོག་པདྨ་བཅུ་གསུམ་པ་དེའི་ནང་ལ་མི་འཇེད་འཇིག་རྟེན་གྱི་ཁམས་ཞེས་པ་ཞིག་ཡོད་པ་དེ་ནི། ཕལ་ཆེར་ད་ལྟ་ང་རང་ཚོ་འགྲོ་འདུག་བྱེད་སའི་ས་ཆ་འདི་ཉིད་ཡིན་ཞེས་གསུང་གི་ཡོད་ས་རེད། མི་འཇེད་འཇིག་རྟེན་གྱི་ཁམས་ཞེས་པ་དེའང་། ད་ལྟ་ང་རང་ཚོའི་བསམ་བློའི་ནང་ལ་ཡོད་པའི་འཛམ་བུ་གླིང་ཟེར་བའི་མཛུབ་མོ་འཛུགས་ས་ཡོད་པ་དེ་འདྲ་ཙམ་ཞིག་མ་ཡིན་པ། གླིང་བཞི་གླིང་ཕྲན་བྱེ་བ་བསམ་གྱིས་མི་ཁྱབ་པ་དེ་འདྲ་ཡིན་ས་རེད། དེ་ཚོ་ཐམས་ཅད་ཀྱི་ནང་ནས། ད་ལྟ་ང་རང་ཚོས་འཛམ་བུ་གླིང་ཞེས་མཛུ་གུ་འཛུགས་ས་ཡོད་པ་འདི་ལའང་། ཚུར་མཐོང་སོ་སོའི་སྐྱེ་བོའི་ངོ་ལ་རྒྱལ་ཁབ་འདྲ་མི་འདྲ་ཉིས་བརྒྱ་ལྷག་ཙམ་དང་། སྐད་རིགས་མི་འདྲ་བ་བདུན་སྟོང་ལྷག་དང་། མི་རིགས་མི་འདྲ་བ་སུམ་ཅུ་ལྷག་དང་། དེ་བཞིན་དུ་གྲོང་ཁྱེར་ཆེན་པོ་ཝ་ར་ན་སི་ཞེས་པ་འདྲ་དང་། ཧྲང་ཧེ། རོམ། ཨི་ས་ཕན་(Isphan)། རྗེ་རུ་ས་ལེམ་(Jerusalem) ཞེས་པ་ལྟ་བུ་རིག་གཞུང་གིས་ཕྱུག་པའི་གྲོང་ཁྱེར་དང་། གྲོང་བརྡལ་མང་པོ། ཨ་རིའི་ནིའི་ཡོག །དབྱིན་ཇི་ལུང་པ་འདྲ་ཡིན་ན་ལོན་ཌོན། སིངྒ་པུར་ལྟ་བུ་དཔལ་འབྱོར་གྱིས་ཕྱུག་པའི་གྲོང་ཁྱེར་དང་གྲོང་བརྡལ་མང་པོ། ལྷ་ས་དང་། ཁ་ཆེའི་ཡུལ་གྱི་མེ་ཁ་(Mecca)། ཡོ་རོབ་ཀྱི་ཝཻ་ཊི་ཀན་ལྟ་བུ་ཆོས་ལུགས་ཀྱི་བསྟི་གནས་ཀྱང་མང་པོ། དེ་ག་ནང་བཞིན་འབར་མ་ལ་ཡོད་པའི་ཤི་ཝ་ཏི་གྷཾ(Shwedagon)། རྒྱ་ནག་གི་གླང་ཆེན་འགྱིང་རི། ཤྲི་ལངྐའི་ཤྲི་དྷ་ལ་དཱ་མ་ལི་ག་བ་(Sri DaladaMaligawa)། བལ་ཡུལ་གྱི་འཕགས་པ་ཤིང་ཀུན་ལྟ་བུ། སངས་རྒྱས་དང་བྱང་ཆུབ་སེམས་དཔའ་གྲུབ་པ་བརྙེས་པའི་གང་ཟག་མང་པོས་ཞབས་ཀྱིས་བཅག་ནས་བྱིན་གྱིས་བརླབས་པའི་གནས་ངོ་མཚར་བ་གྲངས་ལས་འདས་པ་དེ་འདྲ་ཡོད་ནའང་། ད་ལམ་གྱི་གནས་ཕུན་སུམ་ཚོགས་པ་འདི་ནི། སངས་རྒྱས་རེ་རེ་ཙམ་ཕེབས་པའི་ས་ཆ་དེ་འདྲ་མ་ཡིན་པར། བསྐལ་པ་བཟང་པོའི་སངས་རྒྱས་སྟོང་རྩ་གཉིས་ཕེབས་པའི་ས་ཆ། ད་ལྟའི་བར་དུ་སངས་རྒྱས་བཞི་ཕེབས་ཚར་ནས་མ་འོངས་པ་ལའང་སངས་རྒྱས་དགུ་བརྒྱ་དགུ་བཅུ་ལྷག་ཕེབས་པར་འགྱུར་བའི་ས་ཆ། དམིགས་བསལ་གྱི་ང་རང་ཚོའི་སྟོན་པ་ཤཱཀྱ་ཐུབ་པས་བསྐལ་བ་གྲངས་མེད་གསུམ་དུ་ཚོགས་བསགས། སྲིད་པ་མཐའ་མ་རྒྱལ་བའི་སྲས་གཞོན་ནུ་དོན་འགྲུབ་ཀྱིས་སྐྱེ་རྒ་ན་འཆི་ཡི་སྡུག་བསྔལ་ལ་གཟིགས། དེའི་རྐྱེན་གྱིས་ཁྱིམ་ནས་ཁྱིམ་མེད་པ་ལ་རབ་ཏུ་བྱུང་། ཆུ་བོ་ནེ་རཉྫ་ནའི་དྲུང་དུ་ལོ་དྲུག་ཏུ་དཀའ་བ་སྤྱད། བདུད་དཔུང་མང་པོ་སྟན་ཐོག་གཅིག་ཏུ་བཏུལ་བའི་ས་ཆ་ག་ལ་བ། མངོན་པར་རྫོགས་པར་སངས་རྒྱས་པའི་གནས་མཆོག་ས་ཆ་ག་ལ་བ་དེ་ལ་རྡོ་རྗེ་གདན་ཞེས་མིང་གིས་བཏགས། ང་རང་ཚོ་ཐམས་ཅད་ཡོང་སྡོད་སའི་ས་ཆ་འདི་ནི་ད་ལམ་གྱི་གནས་ཕུན་སུམ་ཚོགས་པ་ལགས་སོ། །

གནས་ཕུན་སུམ་ཚོགས་པ་དེ་ལ་བྱོན་པའི་སྟོན་པ་ཕུན་སུམ་ཚོགས་པ་དེ་ནི། ཐེག་པ་ཆེན་པོ་ངེས་དོན་ཡིན་ལུགས་ཀྱི་དབང་དུ་བྱས་ནས་བཤད་བཞག་པ་ཡིན་ན། ང་རང་ཚོའི་སྟོན་པ་དེ་གང་ཟག་གྱི་ན་བ་སོ་སོའི་སྐྱེ་བོ་རིམ་ཅན་པ། དང་པོ་བྱང་ཆུབ་མཆོག་ཏུ་ཐུགས་བསྐྱེད། བར་དུ་བསྐལ་པ་གྲངས་མེད་གསུམ་དུ་ཚོགས་བསགས། མཐའ་མར་མངོན་པར་རྫོགས་པར་སངས་རྒྱས་པའི་གང་ཟག་དེ་འདྲ་གཅིག་ལ་བཤད་རྒྱུ་མ་ཡིན་པར། ཐོག་མ་མེད་པའི་དུས་ནས་ཆོས་ཉིད་འོད་གསལ་གྱི་ཞིང་ཁམས་སུ་མངོན་པར་རྫོགས་པར་སངས་རྒྱས། སྐུ་དང་ཡེ་ཤེས་འདུ་འབྲལ་མེད་པའི་ངང་དུ། སངས་རྒྱས་ཐམས་ཅད་དང་དགོངས་པ་རོ་གཅིག་ཏུ་དབྱེ་བ་མེད་པ་བྱས་ནས་བཞུགས་པའི་ངང་དེ་ལས། དུས་ཚོད་སྐར་མ་གཅིག་ཀྱང་མ་གཡོས་པར། ས་སྟོང་འདྲ་ཞིག་ལ་མཚོན་ན་ཨིན་ཅི་གཅིག་ལས་ཀྱང་མ་གཡོས་པར་། བསམ་གྱིས་མི་ཁྱབ་པར་ལྷུན་གྱིས་གྲུབ་པའི་ངང་ནས་འགྲོ་བའི་དོན་ཆེན་པོ་གནང་དང་གནང་བཞིན་པ་དེ་འདྲ་ཡིན་ས་རེད་དེ། ཡིན་ནའང་ད་ལྟ་གོང་དུ་ཞུས་པ་ནང་བཞིན་ང་ཚོ་འདྲ་བོའི་བསམ་བློའི་ནང་ལ་ཤོང་དཀའ་ལས་ཁག་པོ་ཡིན་ས་རེད། དེ་ཡིན་དུས་ང་ཚོ་འདྲ་བོའི་བློ་ལ་ཤོང་བའི་ཆེད་དུ་སྤྲུལ་པའི་སྐུ་སྣ་ཚོགས་པ་དེ་འདྲ་མང་པོ་བྱུང་བའི་ནང་ནས། ལུམྦི་ནི་ལ་སྐུ་འཁྲུངས། རྡོ་རྗེ་གདན་དུ་མངོན་པར་རྫོགས་པར་སངས་རྒྱས། ཝ་ར་ན་སི་ལ་ཆོས་ཀྱི་འཁོར་ལོ་བསྐོར། མཐར་རྩྭ་མཆོག་གྲོང་དུ་མྱ་ངན་ལས་འདས་པའི་སྤྲུལ་པ་བསྟན་པའི་རྒྱལ་བ་ཤཱཀྱ་ཐུབ་པ་བྱོན། དེའི་རྒྱལ་ཚབ་རྒྱ་གར་འཕགས་པའི་ཡུལ་དུ་སྟོན་པའི་གཏད་རབས་བདུན། གནས་བརྟན་ཆེན་པོ་བཅུ་དྲུག ། འཛམ་གླིང་མཛེས་པར་བྱེད་པའི་རྒྱན་དྲུག ། མཆོག་གཉིས། གྲུབ་པ་ཐོབ་པའི་གྲུབ་ཆེན་བརྒྱད་ཅུ་རྩ་བཞི་ལ་སོགས་པ་དེ་འདྲ་མང་པོ་བྱོན་ཡོད་ས་རེད། དེ་ནས་རྒྱ་གར་ཙམ་ཞིག་མ་ཡིན་པར་ཡུལ་གྲུ་གཞན་ལའང་སྟོན་པའི་རྒྱལ་ཚབ་བསམ་གྱིས་མི་ཁྱབ་པ་ཧ་ཅང་མང་པོ་དེ་འདྲ་བྱོན་ཡོད་ས་རེད། དཔེར་ན་མ་ཧཱ་ཙི་ན་ལྟ་བུའི་རྒྱལ་ཁབ་དང་། ཐེ་ལན། ཤྲི་ལངྐ། སོག་པོ། ཉི་ཧང་དེ་ཚོ་ལའང་རིག་པ་འཛིན་པ་དང་འཛིན་མ། བྱང་ཆུབ་སེམས་དཔའ་དང་སེམས་མ། དགྲ་བཅོམ་པ་དང་། བསྟན་འཛིན་གྱི་སྐྱེས་བུ་བསམ་གྱིས་མི་ཁྱབ་པ་བྱོན་ཡོད་པ་ནི་གོར་མ་ཆགས་པ་ཡིན་རྒྱུ་རེད་བསམ་གྱི་ཡོད། ང་རང་ཚོ་བོད་ཁ་བ་ཅན་གྱི་ཡུལ་དུ་སྟོན་པའི་རྒྱལ་ཚབ་བྱང་ཆུབ་སེམས་དཔའ་ས་ལ་བཞུགས་པ་དང་མ་བཞུགས་པ་མང་པོ་བྱོན་ཡོད་པ་ནི་སྨོས་མ་དགོས་པ་རེད། ད་ལམ་གྱི་སྐབས་སུ་བབ་པའི་ང་རང་ཚོའི་སྟོན་པ་ཕུན་སུམ་ཚོགས་པ་ནི། རྒྱ་གར་འཕགས་པའི་ཡུལ་དུ་ངོ་མཚར་ཆེ་བའི་གདུང་བརྒྱུད་དང་གདུང་རབས་མང་པོ་བྱུང་ཡོད་ནའང་། ནང་པ་སངས་རྒྱས་པ་ཡིན་པའི་ངོས་ནས་མང་པོས་བཀུར་བའི་གདུང་རབས་ཞེས་པ་དེ་ལ་ཧ་ཅང་རྩ་ཆེན་པོ་ཞིག་ཏུ་བརྩི་ཡི་ཡོད་རེད། དཔེ་དེ་བཞིན་དུ་ང་ཚོ་བོད་གངས་ཅན་གྱི་ལྗོངས་ལའང་། ངོ་མཚར་ཆེ་བའི་གདུང་བརྒྱུད་ཀྱི་རིགས་གྲངས་ལས་འདས་པ་མང་པོ་བྱུང་ཡོད་ནའང་། ང་ཚོ་མང་པོ་ཞིག་ལ་ལྟོས་ན་དམིགས་བསལ་གྱི་རྩ་ཆེ་བའི་འོད་གསལ་ལྷ་ཡི་རིགས། འཁོན་གྱི་རིགས། དཔལ་ལྡན་ས་སྐྱ་པ་ཞེས་མཚན་མཆོག་སུམ་ལྡན་གྱི་གྲགས་དཀར་ས་སྟེང་དུ་ཁྱབ་པའི་གདུང་བརྒྱུད་དྲི་མ་མེད་པ། གདུང་བརྒྱུད་དེའི་ནང་དུ་སྔོན་ཡང་འཁོན་ཀླུ་དབང་སྲུང་བ། འཁོན་དཀོན་མཆོག་རྒྱལ་པོ། ཀུན་དགའ་སྙིང་པོ། བསོད་ནམས་རྩེ་མོ། གྲགས་པ་རྒྱལ་མཚན། ས་སྐྱ་པཎྜི་ཏ་ལ་སོགས་པ་བྱང་ཕྱོགས་མུན་པའི་གླིང་གི་སྒྲོན་མེ་གཅིག་རྗེས་གཉིས་མཐུད་དུ་རྒྱུན་མ་ཆད་པ་བྱོན་ཡོད་རེད། ད་ལྟ་ང་རང་ཚོའི་སྐྱབས་མཆོག་རྡོ་རྗེ་འཆང་འདི་ཉིད་ཀྱང་གདུང་བརྒྱུད་རིན་པོོ་ཆེ་དེ་ག་རང་ནས་བྱོན་པ། དེ་ལྟར་བྱོན་པའི་སྟོན་པ་ཕུན་སུམ་ཚོགས་པ་འདི་ལའང་སྟོན་པ་ཕུན་སུམ་ཚོགས་པའི་མཚན་ཉིད་སྤངས་བ་དང་རྟོགས་པ། མཁས་པ་དང་གྲུབ་པ། བཙུན་པ་དང་བཟང་པོའི་ཡོན་ཏན་བསམ་གྱིས་མི་ཁྱབ་པས་བརྒྱན། གཞན་དྲིང་མི་འཇོག་པའི་བདག་ཉིད་ཡིན་པ་ནི་ལྐོག་གྱུར་དུ་མ་སོང་བར་དངོས་སུ་མཐོང་རྒྱུ་དང་། ཐོས་རྒྱུ། ཉམས་སུ་མྱོང་རྒྱུ་ཡོད་པ། སྤྱིར་རྒྱལ་བའི་བསྟན་པ་རིན་པོ་ཆེ་སྣུམ་ཟད་མར་མེ་ལྟ་བུར་གྱུར་པའི་སྐབས་འདི་དང་། ཁྱད་པར་དུ་རྗེ་བཙུན་ས་སྐྱ་པའི་བསྟན་པ་རིན་པོ་ཆེ་ཤིན་ཏུ་ཉམས་འགྲོ་བའི་སྐབས་འདི་ལ་བསྟན་པའི་སྲོག་ཤིང་། བསྟན་པ་རིན་པོ་ཆེའི་དཔུང་གཉེན་དུ་གྱུར་པ། ཡོངས་རྫོགས་བསྟན་པའི་བདག་པོ་དཔལ་ལྡན་ས་སྐྱ་པ་ཞེས་དང་། ཡང་ན་ནི་བྱང་ཕྱོགས་སྐྱོང་བའི་རྒྱལ་པོ་ཞེས་པའི་མཚན་སྙན་གྱི་ཅོད་པཎ་ཅན་ཁ་ཡོད་ལག་ཡོད། དངོས་སུ་མཛུབ་མོ་བཙུགས་ན་འཛུགས་ས་གཅིག་པུ་འདི་ནི། ང་རང་ཚོ་ཐམས་ཅད་ཀྱི་སྐྱབས་དང་རེ་ས་མགོན་དང་དཔུང་གཉེན་དུ་གྱུར་པ་ད་ལྟ་ཞལ་དངོས་སུ་ཞལ་བཞུགས་ཤིང་། ལས་དང་སྐལ་བ་ལྡན་པའི་གདུལ་བྱ་གྲངས་ལས་འདས་པ། ཐུབ་བསྟན་རིས་སུ་མ་ཆད་པའི་ཆོས་བརྒྱུད་འཛིན་པ་མང་པོ། ཁྱད་པར་དུའང་དཔལ་ལྡན་ས་ངོར་ཚར་གསུམ་ཕོ་བྲང་ཁག་ཐམས་ཅད་ཀྱི་གདུང་སྲས་དང་རྗེ་བཙུན་མ་ད་ལྟ་བཞུགས་ཡོད་ཚད་ཐམས་ཅད་ཀྱི་རྩ་བའི་བླ་མར་གྱུར་པ། རྒྱུད་སྡེ་རིན་པོ་ཆེ་དཀྱིལ་འཁོར་རྒྱ་མཚོ་མང་པོའི་དབང་ཡང་ཞུ་ས། མདོ་རྒྱུད་ཐམས་ཅད་ཀྱི་ལུང་དང་ཁྲིད་ཞུ་ས། འདི་ཕྱི་བར་དོ་ཀུན་ཏུ་སྐྱབས་ཞུ་ས་ཉག་གཅིག་ཏུ་གྱུར་པ། རྡོ་རྗེ་རྒྱལ་པོ་ཁྱབ་བདག་ཧེ་རུ་ཀ་དངོས། མཚན་ཙམ་ཡང་བརྗོད་པར་དཀའ་ཡང་དོན་གྱི་སླད་དུ་མཚན་ནས་སྨོས་ན་༸སྐྱབས་མགོན་བདག་ཁྲི་རྡོ་རྗེ་འཆང་ངག་དབང་ཀུན་དགའ་ཐེག་ཆེན་དཔལ་འབར་ཕྲིན་ལས་བསམ་འཕེལ་དབང་གི་རྒྱལ་པོ་ཨ་ཡུར་བཛྲ་རིན་པོ་ཆེ་ད་ལྟ་དངོས་སུ་འཚོ་ཞིང་བཞུགས་ཡོད་པ་འདི་སྐབས་སུ་བབས་པའི་སྟོན་པ་ཕུན་སུམ་ཚོགས་པ་ལགས་སོ། །

ཕུན་སུམ་ཚོགས་པ་གསུམ་པ་དུས་ཕུན་སུམ་ཚོགས་པ་དེ་གང་འདྲ་ཞིག་རེད་ཅེ་ན། གོང་དུ་ཞུས་ཟིན་པ་ནང་བཞིན་དོན་དམ་པ་ལ་དུས་ཞེས་པ་དེ་གཏན་ནས་གཞི་གྲུབ་ཡོད་ས་མ་རེད་དེ། ཡིན་ནའང་སོ་སོའི་སྐྱེ་བོ་ལས་དང་ཉོན་མོངས་པ་འཆིང་བ་ཀུན་ལྡན་ང་ཚོ་འདྲ་བོ་ལ། དུས་ཀྱི་ཐ་སྙད་དང་། དུས་ཀྱི་སྐད་ཆ་ཞིག་མ་བཤད་རང་བཤད་ཡིན་དུས། ད་ལན་གྱི་དུས་ཕུན་སུམ་ཚོགས་པ་འདི་ནི། སྔོན་བསྐལ་པ་མང་པོའི་གོང་དུ་མཐའ་ཡས་པའི་རྒྱ་མཚོ་ཆེན་པོའི་ནང་ལ། གསེར་གྱི་པདྨ་སྟོང་རྩ་གཉིས་དུས་གཅིག་ཏུ་ཁ་བྱེ་བའི་དགེ་མཚན་མངོན་སུམ་དུ་མཐོང་ཡོང་དུས། ལྷ་ཡི་བུ་དང་བུ་མོ་ཐམས་ཅད་ཧ་ལས་ནས། ཨེ་མ་བསྐལ་པ་ནི་བཟང་ངོ་ཞེས་ཆེད་དུ་བརྗོད་པ་ལ་བརྟེན་ནས། ད་ལྟའི་བསྐལ་བ་འདི་ལ་བསྐལ་པ་བཟང་པོ་ཞེས་པའི་མིང་ཏུ་ཐོག་པ། བསྐལ་པ་བཟང་པོ་འདིའི་ནང་ནས་རྩོད་ལྡན་གྱི་གདུལ་བྱ་ལ་དམིགས་བསལ་གྱི་ཐུགས་རྗེ་ཡི་གོ་ཆ་བཙན་པོ་བཞེས་པའི་ཤཱཀྱ་ཐུབ་པ་མི་འཇེད་འཇིག་རྟེན་གྱི་ཁམས་འདི་ལ་ཕེབས། ཁོང་ཕེབས་པ་ཙམ་ཞིག་མ་ཡིན་པར། ཁོང་གིས་ཆོས་འཁོར་ཡང་བསམ་གྱིས་མི་ཁྱབ་པ་བསྐོར། ཆོས་འཁོར་བསྐོར་བ་དེའང་མ་ཉམས་པ་ད་ལྟ་ཡང་ཞལ་བཞུགས་པའི་སྐབས་ཡིན་པ་མ་ཟད། དམིགས་གསལ་ད་ལན་གྱི་དུས་ཚོད་འདི་ནི། སྡེ་བཞི་དགེ་བ་བཅུ་ཚང་བའི་ཡུལ་གྲུ་སྡེ་དགེ་རྒྱལ་ཁབ་ཀྱི་ཆོས་རྒྱལ་བསྟན་པ་ཚེ་རིང་ལྟ་བུ་ཆོས་རྒྱལ་རིམ་བྱོན་རྗེ་འབངས་རྣམས་ཀྱིས། དཔལ་ལྡན་ས་སྐྱ་པ་མཆོད་པའི་གནས་སུ་བཀུར་ནས་མི་ལོ་བརྒྱ་ཕྲག་མང་པོ་སོང་། སྡེ་དགེ་རྗེ་འབངས་རྣམས་ཀྱིས་དམིགས་བསལ་གྱིས་དཔལ་ལྡན་ས་ངོར་གཉིས་ཀྱི་བསྟན་པ་ལ་སྲི་ཞུ་ཡང་རྒྱ་མི་ཆུང་བ་ཞུས་ཡོད་རེད། དེ་ཡིན་དུས་སྡེ་དགེ་རྒྱལ་འབངས་ཀྱི་རིས་སུ་གཏོགས་པའི་ད་ལྟའི་མི་བརྒྱུད་ཐམས་ཅད་ཀྱིས་ཀྱང་། ཕ་མེས་ཀྱི་ཡ་རབས་ཀྱི་བྱ་བཞག་སྙིང་ལ་བཅང་ནས་བསྟན་པ་རིན་པོ་ཆེ་སྤྱི་དང་། ཁྱད་པར་དུ་རྗེ་བཙུན་ས་སྐྱ་པའི་བསྟན་པ་བརྒྱུད་འཛིན་མི་ཉམས་གོང་ནས་གོང་དུ་སྤེལ་བ་དང་། དེའི་རྩ་བ་༸སྐྱབས་མགོན་ཁྲི་ཆེན་རྡོ་རྗེ་འཆང་ཆེན་པོ་མཆོག་སྐུ་ཚེ་ཞབས་པད་ཡུན་རིང་དུ་བརྟན་པའི་ཆེད་དུ་མ་གྲོས་བསམ་པ་གཅིག་མཐུན་གྱིས་བརྟན་བཞུགས་འབུལ་བའི་དུས་ཚོད་ཁྱད་པར་ཅན་འདི་ནི་སྐབས་སུ་བབས་པའི་དུས་ཕུན་སུམ་ཚོགས་པ་ལགས་སོ། །

ཕུན་སུམ་ཚོགས་པ་བཞི་པ་ཆོས་ཕུན་སུམ་ཚོགས་པ་ནི། ཡང་ངེས་པའི་དོན་དུ་སྟོན་པ་ཡང་དག་པར་རྫོགས་པའི་སངས་རྒྱས་ཀྱིས་སྒྲ་ཚིག་ཡི་གེ་བློ་ལས་འདས་པའི་ཆོས་འཁོར་དེ། མཚམས་མཚམས་ལ་ཆོས་ཀྱི་འཁོར་ལོ་བསྐོར་བ་འདྲ་དང་། མཚམས་མཚམས་ལ་ཆོས་ཀྱི་འཁོར་ལོ་མ་བསྐོར་བ་འདྲ་བོ་དེ་འདྲ་མ་ཡིན་པ། རྟག་ཁྱབ་ལྷུན་གྱིས་གྲུབ་པའི་ངོ་བོ་ལ་བསྐོར་དང་སྐོར་བཞིན་པ་ཡིན་ནའང་། གདུལ་བྱ་ང་ཚོ་འདྲ་བོ་གཉིས་འཛིན་ཅན་གྱི་ངོ་ལ། གནས་དང་། དུས། འཁོར་སོ་སོ་ལ་ཆོས་འཁོར་གྱི་རིམ་པའང་འདྲ་མི་འདྲ་མང་པོ་བསྐོར་ཡོད་ཅིང་། ད་ལྟ་ང་རང་ཚོའི་འཛམ་བུ་གླིང་འདིའི་ནང་ལའང་སྤྲུལ་སྐུའི་སྟོན་པ་ཤཱཀྱ་ཐུབ་པས་ཆོས་འཁོར་རིམ་པ་གསུམ་ལ་སོགས་པ་བསྐོར། དེའང་ལུང་དང་རྟོགས་པའི་ཆོས་འཁོར། སྡེ་སྣོད་གསུམ་གྱི་ཆོས་འཁོར། གྲངས་ངེས་རྣམ་གཉིས་ཀྱི་ཆོས་འཁོར་སོགས་ཆོས་ཀྱི་ཕུང་པོ་སྟོང་ཕྲག་བརྒྱད་ཅུ་རྩ་བཞི་བསྐོར་གནང་ཡོད་རེད། མཐར་གདུལ་བྱའི་ལས་དང་སྨོན་ལམ་རྟེན་འབྲེལ་ཕུན་སུམ་ཚོགས་པའི་དབང་གིས་ཆོས་ཕུན་སུམ་ཚོགས་པ་དེ་ཡུལ་གྲུ་མང་པོ་ཞིག་ལ་དར། ང་རང་ཚོ་བོད་ཁ་བ་ཅན་གྱི་ལྗོངས་སུ་ཡང་ཆོས་ཕུན་སུམ་ཚོགས་པ་དེ་བཤད་བརྒྱུད་ཀ་ཆེན་བཅུ། སྒྲུབ་བརྒྱུད་ཤིང་རྟ་ཆེ་བརྒྱད་ལ་སོགས་པ་ལས་བརྒྱུད། རིམ་གྱིས་དར་ནས་ད་ལྟའི་བར་དུ་སྲིད་པའི་རྩ་བ་བདག་འཛིན་མ་རིག་པའི་གཉེན་པོ་ལ་བཞུགས་དང་བཞུགས་བཞིན་པ་རེད།

སྐབས་འདིའི་ཆོས་ཕུན་སུམ་ཚོགས་པ་དེ་སྤྲོས་པ་མེད་པའི་ལྟ་བ། གཡེངས་བ་མེད་པའི་སྒོམ་པ། བྱ་བ་བཏང་བའི་སྤྱོད་པ། སེམས་ཀྱི་གནས་ལུགས་རྟོགས་པའི་འབྲས་བུ། ལྟ་སྒོམ་སྤྱོད་པའི་འབྲས་བུ་ཡིས་ཁོག་ཕུབ། ཁྱད་པར་དུ་གསང་བ་སྔགས་ཀྱི་སྐབས་སུ། འཁོར་བ་ཞེས་པ་མར་སྤང་རྒྱུ་ཞིག་དང་། མྱ་ངན་ལས་འདས་པ་ཞེས་པ་ཡར་བླངས་རྒྱུ་ཡོད་པ་དེ་འདྲའི་ལྟ་བ་དེ་དཔེ་སྡུག་པོ་ཞིག་ལ་བརྩིས་ནས། འཁོར་འདས་དབྱེར་མེད་ཀྱི་ལྟ་བ་སྐྱོང། དེའང་གཞི་འཁོར་བའི་ཆོས་ཐམས་ཅད་སེམས་ཀྱི་མཚན་ཉིད་ཀྱི་ཚུལ་དུ་ཚང་། ལམ་གྱི་ཆོས་ཐམས་ཅད་ཡོན་ཏན་གྱི་ཚུལ་དུ་ཚང་། འབྲས་བུའི་ཆོས་ཐམས་ཅད་ནུས་མཐུ་ཡི་ཚུལ་དུ་སེམས་ཐོག་ན་ཚང་བ་ཡིན་དུས། སེམས་དེ་རྐྱང་པ་ལའང་གཞི་ལམ་འབྲས་གསུམ་ཚང་ཞིང་། སེམས་རྐྱང་པའི་སྟེང་ན་འཁོར་འདས་ཀྱི་ཆོས་ཐམས་ཅད་ཚང་བར་བསྟན་པའི་ལམ་སྣང་བ་གསུམ། རྒྱུད་གསུམ་གྱི་སྒོ་ནས་འཁོར་འདས་དབྱེར་མེད་ཀྱི་ལྟ་བ་གཏན་ལ་ཕབ། ལྟ་བ་དེ་ཉམས་སུ་ལེན་དུས་ཚེ་འདི་ལ་ཞེན་པ་སྤངས། འཁོར་བ་ལ་ཞེན་པ་སྤངས། བདག་དོན་ལ་ཞེན་པ་སྤངས། འཛིན་པ་ཐམས་ཅད་སྤངས་ནས་སྣང་བ་སེམས་སུ་སྒྲུབ། སེམས་སྣང་སྒྱུ་མར་སྒྲུབ། སྒྱུ་མའང་རང་བཞིན་གྱིས་མ་གྲུབ་པར་རིམ་གྱིས་བསྒོམ་པ་ལ་སྦྱར་ནས། མཐའ་མའི་མཐའ་མར་སྲིད་ཞི་གཉིས་ཀྱི་མཐའ་ལ་མི་གནས་པའི་འབྲས་བུ་མྱང་འདས་ཆེན་པོ་ཟུང་འཇུག་ཆོས་སྐུའི་གོ་འཕང་ལ་སྦྱོར་བར་བྱེད་པའི་ལམ་འདི་ནི་སྐབས་སུ་བབ་པའི་ཆོས་ཕུན་སུམ་ཚོགས་པ་ལགས་སོ། །

ཕུན་སུམ་ཚོགས་པ་ལྔ་པ་འཁོར་ཕུན་སུམ་ཚོགས་པ་ནི། སངས་རྒྱས་ཆོས་ཀྱི་སྐུའི་ངོ་བོ་ནས་སྐད་ཆ་བཤད་བཞག་པ་ཡིན་ན། གཙོ་བོ་དང་འཁོར་ཞེས་པ་དེ་འདྲའི་ཐ་སྙད་གཏན་ནས་མ་གྲུབ་པ་ཡིན་ནའང་། སངས་རྒྱས་ལོངས་སྤྱོད་རྫོགས་པའི་སྐབས་སུ་གཙོ་བོ་དང་འཁོར་ངེས་པ་ལྔ་ལྡན་གྱི་ཚུལ་དུ་བཞུགས་ཡོད་པ་དང་། དེ་ནས་གདུལ་བྱ་དམན་པ་ང་ཚོ་འདྲ་བོའི་དོན་དག་ལ་གཙོ་བོ་སྟོན་པ་ཐུབ་པའི་དབང་པོ་འཁོར་ལ་ཤ་རིའི་བུ་དང་། མོངྒལ་གྱི་བུ་ལྟ་བུ་ཉན་ཐོས་ཀྱི་སྡེ་ཆེན་པོ། བྱམས་པ་དང་། སྤྱན་རས་གཟིགས། འཇམ་དབྱངས་ལྟ་བུ་ས་ཆེན་པོ་ལ་གནས་པའི་བྱང་ཆུབ་སེམས་དཔའ་ཐེག་པ་ཆེན་པོའི་དགེ་འདུན་གྱི་སྡེ་གྲངས་ལས་འདས་པ་བཞུགས་དང་བཞུགས་བཞིན་པ། དེ་ནང་བཞིན་ང་རང་ཚོ་བོད་གངས་ཅན་གྱི་ལྗོངས་ལའང་སྟོན་པའི་རྒྱལ་ཚབ་བསྟན་འཛིན་གྱི་སྐྱེས་བུ་མང་པོ་ལ་འཁོར་དང་སློབ་མ་བསམ་གྱིས་མི་ཁྱབ་པ་དེ་འདྲ་བྱུང་དང་འབྱུང་བཞིན་པ་དེ་འདྲ་རེད། ད་ལམ་གྱི་སྐབས་སུ་བབ་པའི་འཁོར་ཕུན་སུམ་ཚོགས་པ་དངོས་དེ། སྐྱབས་མཆོག་འཁོར་ལོའི་མགོན་པོ་བདག་ཁྲི་རྡོ་རྗེ་འཆང་གི་ཐུགས་རྗེ་དང་སྨོན་ལམ་ལས་ཐོན་པའི་རྗེ་བཙུན་ས་སྐྱ་པ་ཡབ་སྲས་ཀྱི་རིང་ལུགས་འཛིན་པ་གོས་དཀར་ལྕང་ལོ་ཅན་གྱི་སྡེ་དང་། ལམ་བསླབ་པ་གསུམ་གྱི་བརྒྱུད་ཕྱུག་པའི་རབ་བྱུང་ངུར་སྨྲིག་གི་སྡེ། འཇམ་དབྱངས་ས་སྐྱ་པའི་འཁོན་གདུང་དྲི་མ་མེད་པ་ཡོངས་རྫོགས། དཔལ་ཨེ་ཝྃ་པའི་མཁན་པོ་དང་ཞབས་དྲུང་ཁག །ཚར་པ་རྡོ་རྗེ་འཆང་གི་རྒྱལ་ཚབ། བླ་མ་སློབ་དཔོན་རྡོ་རྗེ་འཛིན་པ་རྣམས་དང་། རྒྱལ་བའི་བསྟན་པ་རིན་པོ་ཆེ་ལ་དད་པ་དང་མོས་གུས། ལྷག་བསམ་རྣམ་པར་དཀར་བའི་བསྟན་པའི་སྦྱིན་བདག་ཕོ་མོ་མང་པོ། གང་ལྟར་འདི་ན་ཡོད་པའི་ང་ཚོ་ཐམས་ཅད་ཇི་སྲིད་བྱང་ཆུབ་སྙིང་པོ་མ་ཐོབ་ཀྱི་བར་དུ། སྐྱེ་བ་དང་ཚེ་རབས་ཐམས་ཅད་དང་། འདི་ཕྱི་བར་དོ་ཀུན་ཏུ་མགྲིན་གཅིག་དབྱངས་གཅིག་ཏུ། སྐྱབས་མཆོག་འཁོར་ལོའི་མགོན་པོ་དུས་རྟག་ཏུ་སྤྱི་གཙུག་ཏུ་སྒོམ། སྙིང་གི་དཀྱིལ་དུ་སྒོམ། གཏན་གྱི་སྐྱབས་གནས་སུ་འཛིན་པ་ཤ་སྟག་ཡིན་དུས་འཁོར་ཕུན་སུམ་ཚོགས་པ་ལགས་སོ། །

འདི་འདྲ་བའི་ཕུན་སུམ་ཚོགས་པ་ལྔ་དང་ལྡན་པའི་སྐབས་འདི་ལ། རང་གཞན་སེམས་ཅན་ཐམས་ཅད་ཀྱི་གནས་སྐབས་དང་མཐར་ཐུག་གི་ཕན་བདེ་མ་ལུས་པ་འབྱུང་བའི་གཞི་སྒོ་གཅིག་ཏུ་གྱུར་པ་ནི། རྒྱལ་བའི་བསྟན་པ་རིན་པོ་ཆེ་འབའ་ཞིག་ཡིན་པ་རེད། བསྟན་པ་རིན་པོ་ཆེ་འདི་ཉིད་ཕྱོགས་ཀུན་ཏུའམ། གནས་སྐབས་ཀུན་ཏུ་མི་ཉམས་པར་གོང་ནས་གོང་དུ་འཕེལ་དགོས་ཀྱི་ཡོད་པ་ཡིན་ན། སོ་སོའི་སྐྱེ་བོ་ལས་དང་ཉོན་མོངས་པའི་གཞན་དབང་ཅན། ང་འདྲ་བོ་ཡིས་སང་ཉིན་གང་བྱེད་དགོས་ཀྱི་ཡོད་མེད་ཀྱི་བསམ་བློ་ཡང་གཏོང་མི་ཤེས་མཁན། དབང་ཆ་ཡང་མེད་མཁན་ཚོས་འཆར་གཞི་བཀོད། སྒྲིག་འཛུགས་བྱེད། ཇུས་སྒྲིག་བྱེད་ནས་བསྟན་པ་རིན་པོ་ཆེ་འཛིན་སྐྱོང་སྤེལ་གསུམ་བྱེད་ཅེས་པ་དེ་ཧ་ཅང་གི་དཀའ་ལས་ཁག་པོ་ཡིན་ས་རེད།

དེ་ནང་བཞིན་དཔལ་འབྱོར་དང་། དཔུང་ཤུགས། དབང་ཆ། སྲིད་ཇུས་ཞེས་པ་དེ་ཚོས་ཀྱང་བསྟན་པ་འཛིན་མི་ནུས། ང་ཚོ་ལས་ལྷག་མེད་བསམ་པའི་ཁ་ཕོ་རུས་དང་ཞེན་ཁོག་གིས་ཀྱང་བསྟན་པ་སྐྱོང་མི་ནུས། ཡང་མཚམས་མཚམས་ལ་ང་ཚོ་དཔལ་འབྱོར་དང་ནུས་སྟོབས་སོགས་གང་ཡང་མེད་བསམ་ནས་སྡུག་ཡུས་བཤད་བཞག་པ་དེས་ཀྱང་བསྟན་པ་འཛིན་པ་དེ་དཀའ་ལས་ཧ་ཅང་ཁག་པོ་ཡིན་ས་རེད། དེ་ཡིན་དུས་ད་ང་རང་ཚོས་བཟོས་ནའང་བཟོ་ཐུབ། བཟོས་ན་ཕན་ཐོག་ཀྱང་ཡོད་པ། ལས་ཀ་རྒྱ་ཆེན་པོ་རེད་དེ། ཡིན་ནའང་། དཀའ་ལས་མང་པོ་རྒྱག་མི་དགོས་པ། བསྟན་པ་དང་འགྲོ་བའི་ཕན་བདེའི་རྩ་བ་དེ་སྐྱབས་མཆོག་རྡོ་རྗེ་འཆང་ཁོ་རང་རང་ལ་ཐུག་བསྡད་པ་ཡིན་དུས། སྐྱབས་མཆོག་འཁོར་ལོའི་མགོན་པོ་སྐུ་ཚེ་མི་ཤིགས་རྡོ་རྗེ་ཆོས་བདུན་གྱི་ངོ་བོར་ཞབས་པད་ཡུན་རིང་དུ་བརྟན་ཡོད་པ་ཡིན་ན། དེ་རྐྱང་པ་མ་གཏོགས་དེ་ལས་གཞན་ལས་ཀ་གང་ཡང་འབད་བརྩོན་བྱེད་དགོས་པ་གཏན་ནས་མེད་པ། དེ་རང་གིས་འགྲིག་སྡོད་པ་ཡིན་དུས། འདི་ན་ཡོད་པ་ང་ཚོ་ཐམས་ཅད་ཀྱིས། གཙོ་བོ་དད་པ་དང་དག་སྣང་། དམ་ཚིག་སྙིང་ལ་བཅང་། སྐྱབས་མཆོག་འཁོར་ལོའི་མགོན་པོ་སྐུ་ཚེ་ཡུན་རིང་དུ་ཞབས་པད་བརྟན་པའི་གསོལ་བ་འདེབས་པའི་ཆེད་དུ། མཆོད་པའི་མཎྜལ་འབུལ་བ་ལ། མདོ་ནང་བཞིན་བྱས་པ་ཡིན་ན། གླིང་བཞི་གླིང་ཕྲན་བྱེ་བ་ཕྲག་བརྒྱ་ལ་སོགས་པ་རབ་འབྱམས་རྒྱལ་བའི་ཞིང་ཀུན་མཆོད་སྤྲིན་གྱིས་ཕྱུར་བུར་བཀང་ནས་འབུལ་རྒྱུ་དང་། གསང་སྔགས་རྡོ་རྗེ་ཐེག་པའི་དབང་དུ་བྱས་པ་ཡིན་ན། སྣང་གྲགས་རིག་གསུམ་ལྷ་སྔགས་ཡེ་ཤེས་ཀྱི་ངོ་བོའམ། ཡང་ན་ལུས་ངག་ཡིད་གསུམ་སྐུ་གསུང་ཐུགས་རྡོ་རྗེ་ཡི་ཞིང་ཁམས་དང་། མདོར་ན་སྣང་སྲིད་སྣོད་བཅུད་ཐམས་ཅད་བདེ་སྟོང་རྡོ་རྗེའི་རོལ་མོར་འཆར་ནས་འབུལ་བ་སོགས་གང་ལྟར་མཎྜལ་གྱི་བྱ་བ་རྐྱང་པ་ལའང་ངོ་བོ་ཕ་རོལ་དུ་ཕྱིན་པ་དྲུག་ཚང་ཞིང་། ཕར་ལྟ་ས་ལ་ང་རང་ཚོ་སོ་སོའི་སྐྱེ་བོའི་བློ་ཡི་ངོར་ག་རེ་རྩ་ཆེན་པོ་ཡིན་པ་སྟེ། གང་ཡག་ཡག ། གང་སྙིང་རྗེ་བོ་དང་རིན་ཐང་ཆེན་པོ་ཅན་དངོས་སུ་འབུལ་ཐུབ་པ་དང་། དངོས་སུ་འབུལ་མ་ཐུབ་ནའང་བློ་ཡིས་སྤྲུལ་ནས་མ་ཚང་བ་མེད་པའམ། ཚང་ཞིང་ཡིད་དུ་འོང་བ་མངོན་པར་སྤྲུལ་ནས་ཐམས་ཅད་ཀྱིས་ལྷན་ཅིག་ཏུ་ཞིང་ཁམས་འབུལ་བར་ཞུ།

སྤྱི་ལོ་ ༢༠༢༢ ཕྱི་ཟླ་ ༡༡ ཚེས་ ༢༤ དང་། རབ་བྱུང་ ༡༧ ཆུ་ཕོ་སྟག་གི་ལོའི་བོད་ཟླ་ ༡༠ ཚེས་༡ཉིན་གནས་མཆོག་རྡོ་རྗེ་གདན་དུ་སྡེ་བཞི་དགེ་བཅུ་ཚང་བའི་ཆོས་རྒྱལ་ཆེན་པོའི་འཁོར་འབངས་སུ་གཏོགས་པའི་དཔལ་བརྩེ་བ་ཆེན་པོའི་རྗེས་འཇུག་པ་ཡོངས་ཀྱིས་༸སྐྱབས་མགོན་ས་སྐྱ་གོང་མ་ཁྲི་ཆེན་རྡོ་རྗེ་འཆང་ཨ་ཡུ་བཛྲ་རིན་པོ་ཆེ་མཆོག་ལ་ལོ་རྒྱུས་རང་བཞིན་ཅན་གྱི་བརྟན་བཞུགས་འབུལ་བའི་སྐབས་སུ། སྐྱབས་རྗེ་རྫོང་སར་འཇམ་དབྱངས་མཁྱེན་བརྩེ་ཐུབ་བསྟན་ཆོས་ཀྱི་རྒྱ་མཚོས་ལྗགས་བཤད་གནང་བའི་ཕུན་ཚོགས་ལྔ་ལྡན་གྱི་བཤད་པ་དད་པའི་སྤུ་ལོང་གཡོ་བའི་གསུང་དབྱངས་ཡན་ལག་ལྔ་ལྡན་ཞེས་བྱ་བ་འདི། ལྗགས་བཤད་གནང་བའི་སྒྲ་སྒམ་དང་གཟུགས་མཐོང་བརྙན་ཕྲིན་ལས་ཡིག་ཐོག་ཏུ་ཕབ། མིང་ཡང་བཏགས་ནས་རིན་པོ་ཆེ་མཆོག་ལ་ཞུ་དག་དང་གནང་བ་བཅས་ཞུས་ཏེ་སྤེལ་བ་ཡིན་པས། དཔལ་ལྡན་བླ་མ་དམ་པའི་སྐུ་ཚེ་ཞབས་པད་རྒྱལ་བ་ཚེ་མཐའ་ཡས་པའི་ངོ་བོར་བརྟན་པའི་རྒྱུར་གྱུར་ཅིག་ཅེས་ཡི་གེར་ཕབ་སྒྲིག་པ་གཟི་བརྗིད་མཐའ་ཡས་ཀྱིས་རྫོང་སར་བཤད་གྲྭ་ཆོས་ཀྱི་བློ་གྲོས་ནས་སྨོན་པའོ།། །།སརྦ་མངྒ་ལཾ།། །།
